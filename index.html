nexus-workspace.config 
import { AI_Engine } from '@nexus/core';

// Menghubungkan output AI ke Live Link
const project = new AI_Engine({
  framework: 'Multipage_HTML5',
  ssl: true,
  edge_hosting: true
});

project.deploy().then((url) => {
  console.log(`Aplikasi sedia di: ${url}`);
});
